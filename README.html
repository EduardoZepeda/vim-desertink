 31   " functions {{{
 32   " returns an approximate grey index for the given grey level
 33   fun! <SID>grey_number(x)
 34     if &t_Co == 88
 35       if a:x < 23
 36         return 0
 37       elseif a:x < 69
 38         return 1
 39       elseif a:x < 103
 40         return 2
 41       elseif a:x < 127
 42         return 3
 43       elseif a:x < 150
 44         return 4
 45       elseif a:x < 173
 46         return 5
 47       elseif a:x < 196
 48         return 6
 49       elseif a:x < 219
 50         return 7
 51       elseif a:x < 243
 52         return 8
 53       else
 54         return 9
 55       endif
 56     else
 57       if a:x < 14
 58         return 0
 59       else
 60         let l:n = (a:x - 8) / 10
 61         let l:m = (a:x - 8) % 10
 62         if l:m < 5
 63           return l:n
 64         else
 65           return l:n + 1
 66         endif
 67       endif
 68     endif
 69   endfun
 70 
 71   " returns the actual grey level represented by the grey index
 72   fun! <SID>grey_level(n)
 73     if &t_Co == 88
 74       if a:n == 0
 75         return 0
 76       elseif a:n == 1
 77         return 46
 78       elseif a:n == 2
 79         return 92
 80       elseif a:n == 3
 81         return 115
 82       elseif a:n == 4
 83         return 139
 84       elseif a:n == 5
 85         return 162
 86       elseif a:n == 6
 87         return 185
 88       elseif a:n == 7
 89         return 208
 90       elseif a:n == 8
 91         return 231
 92       else
 93         return 255
 94       endif
 95     else
 96       if a:n == 0
 97         return 0
 98       else
 99         return 8 + (a:n * 10)
100       endif
101     endif
102   endfun
103 
104   " returns the palette index for the given grey index
105   fun! <SID>grey_color(n)
106     if &t_Co == 88
107       if a:n == 0
108         return 16
109       elseif a:n == 9
110         return 79
111       else
112         return 79 + a:n
113       endif
114     else
115       if a:n == 0
116         return 16
117       elseif a:n == 25
118         return 231
119       else
120         return 231 + a:n
121       endif
122     endif
123   endfun
124 
125   " returns an approximate color index for the given color level
126   fun! <SID>rgb_number(x)
127     if &t_Co == 88
128       if a:x < 69
129         return 0
130       elseif a:x < 172
131         return 1
132       elseif a:x < 230
133         return 2
134       else
135         return 3
136       endif
137     else
138       if a:x < 75
139         return 0
140       else
141         let l:n = (a:x - 55) / 40
142         let l:m = (a:x - 55) % 40
143         if l:m < 20
144           return l:n
145         else
146           return l:n + 1
147         endif
148       endif
149     endif
150   endfun
151 
152   " returns the actual color level for the given color index
153   fun! <SID>rgb_level(n)
154     if &t_Co == 88
155       if a:n == 0
156         return 0
157       elseif a:n == 1
158         return 139
159       elseif a:n == 2
160         return 205
161       else
162         return 255
163       endif
164     else
165       if a:n == 0
166         return 0
167       else
168         return 55 + (a:n * 40)
169       endif
170     endif
171   endfun
172 
173   " returns the palette index for the given R/G/B color indices
174   fun! <SID>rgb_color(x, y, z)
175     if &t_Co == 88
176       return 16 + (a:x * 16) + (a:y * 4) + a:z
177     else
178       return 16 + (a:x * 36) + (a:y * 6) + a:z
179     endif
180   endfun
181 
182   " returns the palette index to approximate the given R/G/B color levels
183   fun! <SID>color(r, g, b)
184     " get the closest grey
185     let l:gx = <SID>grey_number(a:r)
186     let l:gy = <SID>grey_number(a:g)
187     let l:gz = <SID>grey_number(a:b)
188 
189     " get the closest color
190     let l:x = <SID>rgb_number(a:r)
191     let l:y = <SID>rgb_number(a:g)
192     let l:z = <SID>rgb_number(a:b)
193 
194     if l:gx == l:gy && l:gy == l:gz
195       " there are two possibilities
196       let l:dgr = <SID>grey_level(l:gx) - a:r
197       let l:dgg = <SID>grey_level(l:gy) - a:g
198       let l:dgb = <SID>grey_level(l:gz) - a:b
199       let l:dgrey = (l:dgr * l:dgr) + (l:dgg * l:dgg) + (l:dgb * l:dgb)
200       let l:dr = <SID>rgb_level(l:gx) - a:r
201       let l:dg = <SID>rgb_level(l:gy) - a:g
202       let l:db = <SID>rgb_level(l:gz) - a:b
203       let l:drgb = (l:dr * l:dr) + (l:dg * l:dg) + (l:db * l:db)
204       if l:dgrey < l:drgb
205         " use the grey
206         return <SID>grey_color(l:gx)
207       else
208         " use the color
209         return <SID>rgb_color(l:x, l:y, l:z)
210       endif
211     else
212       " only one possibility
213       return <SID>rgb_color(l:x, l:y, l:z)
214     endif
215   endfun
216 
217   " returns the palette index to approximate the 'rrggbb' hex string
218   fun! <SID>rgb(rgb)
219     if a:rgb == "ffffff"
220       return 15
221     endif
222 
223     let l:r = ("0x" . strpart(a:rgb, 0, 2)) + 0
224     let l:g = ("0x" . strpart(a:rgb, 2, 2)) + 0
225     let l:b = ("0x" . strpart(a:rgb, 4, 2)) + 0
226 
227     return <SID>color(l:r, l:g, l:b)
228   endfun
229 
230   " sets the highlighting for the given group, add optional fg, bg and attr
231   " values to override the terminal settings
232   fun! <SID>X(group, fg, bg, attr, ...)
233     let l:ctermfg   = (a:0 > 0 && a:1 != "") ? a:1 : <SID>rgb(a:fg)
234     let l:ctermbg   = (a:0 > 1 && a:2 != "") ? a:2 : <SID>rgb(a:bg)
235     let l:ctermattr = (a:0 > 2 && a:3 != "") ? a:3 : a:attr
236 
237     if a:fg != ""
238       exec "hi " . a:group . " guifg=#" . a:fg . " ctermfg=" . l:ctermfg
239     endif
240     if a:bg != ""
241       exec "hi " . a:group . " guibg=#" . a:bg . " ctermbg=" . l:ctermbg
242     endif
243     if a:attr != ""
244       exec "hi " . a:group . " gui=" . a:attr
245     endif
246     if l:ctermattr != ""
247       exec "hi " . a:group . " cterm=" . l:ctermattr
248     endif
249   endfun
